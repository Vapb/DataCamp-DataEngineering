Arrays in bash

array_example=(1 2 3) (Example of array in bash, arrays dont use ,)

declare -a array_example (Creating a empty array)

echo ${array[@]} returns all array

#array[@] returns the length

echo ${my_array[3]} (Returns the third element of my_array)

echo ${my_array[@]:3:2} (Silicing array example N→ Start index , M→ Number of elements to return)

my_array[0]=0 (Overwriting example)

my_array+=(11) (Appending example)

my_array=(1 2 3 4 5)
my_array[0]=0
echo ${my_array[0]}
# Returns 0. Dont neeed to use $ when manipulating one position
my_array+=(10)
# appending example

Associative Arrays in Bash (Python Dictionary)

declare -A my_array (Creating an empty array)

declare -A my_array=([city_name]=”New York” [population]=1000000) (Creating array with items)

echo ${!my_array[@]} (Returns all the keys)

my_array[”key”]=”value” (Adding elements)

# Create variables from the temperature data files
temp_b="$(cat temps/region_B)"
temp_c="$(cat temps/region_C)"

# Create an array with these variables as elements
region_temps=($temp_b $temp_c)

# Call an external program to get average temperature
average_temp=$(echo "scale=2; (${region_temps[0]} + ${region_temps[1]}) / 2" | bc)

# Append to array
region_temps+=($average_temp)

# Print out the whole array
echo ${region_temps[@]}

# Create a function 
function return_percentage () {

  # Calculate the percentage using bc
  percent=$(echo "scale=2; 100 * $1 / $2" | bc)

  # Return the calculated percentage
  echo $percent
}

# Call the function with 456 and 632 and echo the result
return_test=$(return_percentage 456 632)
echo "456 out of 632 as a percent is $return_test%"
